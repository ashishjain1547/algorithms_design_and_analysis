My Code


def solve(N, M, A):
    A = sorted(A)

    n = N
    cnt = 0

    for a in A:
        n -= a
        if n >= 0:
            cnt += 1
        else:
            break
    return cnt



t = int(input())

for _ in range(t):
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    print(solve(N, M, A))
